1""" 

2This module, `kcp_ds` (Kernel Change Point Detect Select), encapsulates functions from the Ruptures 

3package and includes a function for penalized variable selection. 

4 

5Specific details and remarks: 

61. The function outputs Python indices of the columns of "data" corresponding to the detected 

7change points. Real-world times associated with the data are not considered in this function. 

82. The ruptures class KernelCPD has a minimum allowed segment size (min_size). It is currently set 

9to 2 by default and is not an input parameter for kcp_ds. 

103. The default kernel is "linear," with options for "cosine" and "rbf" kernels. To cover all bases, 

11an input params is required, which defaults to None but can be modified for the rbf kernel (e.g., 

12params = {"gamma": value_of_gamma}). 

134. Due to memory constraints, the algorithm processes offline time-series data longer than max_n_time_points 

14in segments of length max_n_time_points, and then outputs the full set of change points. 

15 

16Note: Please handle real-world times associated with the data independently as this module focuses 

17on change point detection and variable selection. 

18""" 

19

31def kcp_ds( 

32 data: np.array, 

33 kernel: Literal["linear", "cosine", "rbf"] = "linear", 

34 params: Optional[Dict[str, Any]] = None, 

35 max_n_time_points: int = 2000, 

36 min_n_time_points: int = 10, 

37 expected_frac_anomaly: float = 1/1000, 

38) -> Tuple[List[int], List[int]]: 

39 """Return a set of important change points change points after 

40 running kernel change point detection followed by penalized model 

41 selection using the slope heuristic method. 

42 

43 Since the criterion is responding to noise/overfitting, penalized 

44 variable selection is performed in order to obtain a final decision 

45 as to how many of the detected change points are truly significant 

46 and not simply adjusted noise.  

47 

48 Args: 

49 data: array of dimension (number of time points) x (number of 

50 time-series) containing the data points. 

51 kernel: Kernel of the kernet change_point detection. If it is 

52 rbf, params must include a "gamma" parameters, with a 

53 positive real value. 

54 params: parameters for the kernel instance 

55 max_n_time_points: maximum size (max_n_time_points) x 

56 (max_n_time_points) of matrices expected to be processed 

57 quickly by the computer system. 

58 min_n_time_points: minimum number of time points in the current 

59 dataset for which it makes sense to run the detection 

60 algorithm again. If there are fewer than min_n_time_points 

61 points in the dataset, no computations will be run and the 

62 outputs will be empty. 

63 expected_frac_anomaly: This parameter encodes prior knowledge  

64 of users as to how often anomalies might occur. Results can 

65 be quite dependent on the choice of this parameter, so  

66 choose carefully! 

67 """ 

68 if max_n_time_points < min_n_time_points: 

69 raise ValueError( 

70 "max_n_time_points should be greater than min_n_time_points." 

71 ) 

72 

73 if expected_frac_anomaly > 1/2: 

74 raise ValueError( 

75 "expected_frac_anomaly should be at most 1/2." 

76 ) 

77 

78 n_total_time_points = np.shape(data)[0] 

79 

80 current_data = np.copy(data) # new variable for internal manipulation 

81 

82 n_current_time_points = np.shape(current_data)[0] 

83 curr_n_remaining_t_points = n_current_time_points 

84 

85 # Initialise an empty list to store the end points of the intervals of length max_n_time_points 

86 # (except for the last interval, which could be shorter). This variable will be required for 

87 # the explainability function that can be run after the current function: 

88 interval_end_points = [] 

89 

90 current_zero = 0 # remember the original Python indices 

91 

92 detected_change_points = [] 

93 

94 while curr_n_remaining_t_points > max_n_time_points: 

95 # Get the block of data to plug into the Ruptures setting: 

96 data_use = current_data[ 

97 :max_n_time_points, 

98 : 

99 ] 

100 

101 # Store the true index of the time point at the end of this interval, which will be needed 

102 # in the explainability function later: 

103 interval_end_points.append(current_zero + max_n_time_points) 

104 

105 detected_change_points, next_ch_pts = _detect_changepoints_from_kernelcpd( 

106 kernel=kernel, 

107 params=params, 

108 data_use=data_use, 

109 expected_frac_anomaly=expected_frac_anomaly, 

110 current_zero=current_zero, 

111 detected_change_points=detected_change_points, 

112 ) 

113 

114 # Concatenate any current data rows left after the last change point found, to the 

115 # remaining data (if there is any left): 

116 current_data = data[(next_ch_pts[-1]-1):, :] 

117 

118 current_zero = next_ch_pts[-1] - 1 

119 

120 n_current_time_points = np.shape(current_data)[0] 

121 curr_n_remaining_t_points = n_current_time_points 

122 

123 # Deal with any residual block of data that is shorter than (or equal to)  

124 #max_n_time_points but still larger than min_n_time_points: 

125 if ( 

126 curr_n_remaining_t_points <= max_n_time_points 

127 ) & ( 

128 curr_n_remaining_t_points >= min_n_time_points 

129 ): 

130 

131 # Store the true index of the time point at the end of this interval, which will be needed 

132 # in the explainability function later. Unlike in the above while loop, here the time point 

133 # at the end of the interval is the last time point in the dataset: 

134 interval_end_points.append(n_total_time_points) 

135 

136 detected_change_points, next_ch_pts = _detect_changepoints_from_kernelcpd( 

137 kernel=kernel, 

138 params=params, 

139 data_use=current_data, 

140 expected_frac_anomaly=expected_frac_anomaly, 

141 current_zero=current_zero, 

142 detected_change_points=detected_change_points, 

143 ) 

144 

145 #We need to remove any "detected" change-points which were actually ends of intervals. 

146 #First we add 1 to each detected_change_point: 

147 temp_detec = [detected_change_points[i] + 1 for i in range(len(detected_change_points))] 

148 #Now look for matches with values in "interval_end_points" 

149 index_matches = [i for i in range(len(detected_change_points)) if temp_detec[i] in interval_end_points] 

150 detected_change_points = [i for j, i in enumerate(detected_change_points) if j not in index_matches] 

151 

152 return detected_change_points, interval_end_points

155def _detect_changepoints_from_kernelcpd( 

156 kernel: Literal["linear", "cosine", "rbf"], 

157 params: Optional[Dict[str, Any]], 

158 data_use: np.array, 

159 expected_frac_anomaly: float, 

160 current_zero: int, 

161 detected_change_points: List[int], 

162) -> Tuple[List[int], List[int]]: 

163 """Apply kernel change point detection and update 

164 detected_change_points. 

165 

166 Auxiliary function to kcp_ds. Update the detected_change_points 

167 variable in place. 

168 

169 Args: 

170 kernel: Kernel of the kernet change_point detection. If it is 

171 rbf, params must include a "gamma" parameters, with a 

172 positive real value. 

173 params: parameters for the kernel instance 

174 data_use: chunk of data on which to apply the detection 

175 expected_frac_anomaly: This parameter encodes prior knowledge  

176 of users as to how often anomalies might occur. Results can 

177 be quite dependent on the choice of this parameter, so  

178 choose carefully! 

179 current_zero: index of the first point of the current chunk of  

180 the dataset 

181 detected_change_points: all breakpoints detected so far 

182 

183 Returns: 

184 The list of all points detected so far. 

185 The list of points newly detected during the last iteration. 

186 """ 

187 # Associate the data with the class KernelCPD from the Ruptures library: 

188 algo = rpt.KernelCPD(kernel=kernel, params=params) 

189 algo.fit(data_use) 

190 

191 #Get a value for the maximum number of change-points to look for; essentially, 

192 #this is set as = 2 * fraction of expected anomalies * number of current time points. 

193 n_bkps_max = max( 1, int(2*np.floor(expected_frac_anomaly*np.shape(data_use)[0])) ) 

194 

195 array_of_n_bkps = np.arange(1, n_bkps_max + 1) 

196 algo_costs = [ 

197 get_sum_of_cost(algo=algo, n_bkps=n_bkps) 

198 for n_bkps in array_of_n_bkps 

199 ] # value of the criterion for when we have chosen to stick with 1 up to n_bkps_max points 

200 # Since each time we add an extra change point, the criterion will never decrease: 

201 # algo_costs is a non-increasing list. 

202 

203 #For the linear kernel we want to concatenate onto the FRONT of this list the value of the 

204 #criterion when there are ZERO change-points. The problem is that the ruptures package  

205 #assumes that there is at least one change-point, while the model selection of Arlot et al 

206 #(2019) includes the case when there are ZERO change-points. We emphasize that the matrix 

207 #multiplication below is only for the linear kernel. 

208 linear_prod = np.dot(data_use,data_use.T) 

209 diag_sum = np.sum([linear_prod[i,i] for i in range(np.shape(data_use)[0])]) 

210 crit_linear_zero_cp = diag_sum - np.mean(linear_prod) 

211 #Concatenate this value onto the FRONT of the algo_costs list: 

212 algo_costs.insert(0,crit_linear_zero_cp) 

213 

214 n_bkps_pen = _model_select_n_bp( 

215 data=data_use, 

216 algo_costs=algo_costs, 

217 array_of_n_bkps=array_of_n_bkps, 

218 n_bkps_max=n_bkps_max, 

219 ) # penalized selection 

220 

221 # Extract the solution: 

222 if n_bkps_pen > 0: 

223 next_ch_pts = algo.predict(n_bkps=n_bkps_pen) 

224 # Remove the end point, which itself is not an endpoint: 

225 del next_ch_pts[-1] 

226 else: 

227 #If there were NO change-points detected, by convention we shall  

228 #instead output the end-point index of the interval AS IF it were 

229 #a change-point. Note that this (index + 1) was also already stored in 

230 #the variable "interval_end_points" so we will be able to weed out 

231 #this change-point at the end for the final list and for the 

232 #eventual visualization steps. 

233 next_ch_pts = algo.predict(n_bkps=1) 

234 next_ch_pts = [next_ch_pts[-1]-1] 

235 

236 # Add the current zero index (in order to get the true row/time point in the data): 

237 next_ch_pts = [x + current_zero for x in next_ch_pts] 

238 

239 # Concatenate these with any found in previous loops: 

240 detected_change_points = detected_change_points + list(next_ch_pts) 

241 

242 return detected_change_points, next_ch_pts

245def _model_select_n_bp( 

246 data: np.array, 

247 algo_costs: Sequence[float], 

248 array_of_n_bkps: Sequence[int], 

249 n_bkps_max: int, 

250) -> int: 

251 """Calculate the number of change points after model selection using 

252 the slope heuristic method. 

253 

254 This function takes the output set of candidate change points from 

255 the Ruptures library method and refines them to a subset that is 

256 likely to represent signal rather than (over)fitted noise. It 

257 implements the penalized variable selection method described 

258 in Arlot et al. (2019). 

259 

260 Args: 

261 data: signal data points in which to calculate the optimal 

262 number of change points 

263 algo_costs: the (non-penalized) costs associated with each 

264 number of change points 

265 array_of_n_bkps: array listing the number of change points to be 

266 considered 

267 n_bkps_max: maximum number of change points to be detected 

268 

269 Returns: 

270 The number of change points after penalized variable selection 

271 using the slope heuristic method in Arlot et al. (2019). 

272 """ 

273 lower_D = math.ceil(0.6 * n_bkps_max) 

274 

275 X1 = [ 

276 (1/np.shape(data)[0])*math.log(math.comb(np.shape(data)[0] - 1, D - 1)) 

277 for D in np.arange(lower_D, n_bkps_max + 1) 

278 ] 

279 X2 = [(1/np.shape(data)[0])*D for D in np.arange(lower_D, n_bkps_max + 1)] 

280 

281 X = np.transpose(np.array([X1, X2])) 

282 y = np.array([(1/np.shape(data)[0])*algo_costs[D - 1] 

283 for D in np.arange(lower_D, n_bkps_max + 1)]) 

284 # 

285 ## Linear regression: 

286 algo_reg = LinearRegression().fit(X, y) 

287 s1_hat = algo_reg.coef_[0] 

288 s2_hat = algo_reg.coef_[1] 

289 

290 ########################################################### 

291 

292 # Final parameters: 

293 c1 = -2 * s1_hat 

294 c2 = -2 * s2_hat 

295 

296 # Use these two parameters to do the model selection in Arlot et al. (2019):  

297 

298 algo_pen = [ 

299 algo_costs[i-1] 

300 + 

301 c1 * math.log(math.comb(np.shape(data)[0] - 1, i - 1)) 

302 + 

303 c2 * i 

304 for i in array_of_n_bkps 

305 ] 

306 

307 min_val = min(algo_pen) 

308 all_min_indices = [i for i, x in enumerate(algo_pen) if x == min_val] 

309 

310 # There could be more than one items in this object. We choose the smallest for parcimony 

311 # reasons in this (unlikely) event: 

312 min_index = all_min_indices[0] 

313 

314 n_bkps_pen = min_index 

315 

316 return n_bkps_pen
